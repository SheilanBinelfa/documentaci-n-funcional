import streamlit as st
from docx import Document
from docx.shared import Inches
import io
import re

st.set_page_config(page_title="Endalia Doc Engine Pro", layout="wide")

# --- LÓGICA DE DETECCIÓN DE IMAGEN Y SECCIÓN ---
def analizar_documento_maestro(file):
    doc = Document(file)
    texto_completo = "\n".join([p.text for p in doc.paragraphs])
    
    # Buscar la última imagen [Image XX]
    imagenes = re.findall(r'\[Image (\d+)\]', texto_completo)
    ultima_img = int(imagenes[-1]) if imagenes else 97
    
    # Extraer títulos para el selector
    titulos = [p.text for p in doc.paragraphs if p.style.name.startswith('Heading')]
    return ultima_img, titulos

st.title("🚀 Orquestador Inteligente: Planificación y Registro")

archivo_maestro = st.sidebar.file_uploader("Carga el Funcional 25Q4", type=['docx'])

if archivo_maestro:
    # Analizamos el documento real nada más cargarlo
    ultima_img_detectada, lista_titulos = analizar_documento_maestro(archivo_maestro)
    prox_img = ultima_img_detectada + 1

    col_in, col_pre = st.columns([1, 1])

    with col_in:
        st.subheader("📥 Entrada de Nueva Funcionalidad")
        pbi = st.text_area("PBI / Requisitos del Backlog:", height=150, placeholder="Ej: Nueva alerta de olvido de fichaje...")
        capturas = st.file_uploader("Capturas de pantalla:", accept_multiple_files=True)
        
        # IA propone ubicación
        seccion_sugerida = "7.2.6 Notificaciones de Sistema" # Aquí iría la lógica de búsqueda semántica
        ubicacion = st.selectbox("Insertar después de:", lista_titulos, index=len(lista_titulos)-1)
        nombre_nueva = st.text_input("Título de la nueva sección:", value=seccion_sugerida)

    with col_pre:
        st.subheader("👁️ Previsualización Estilo Endalia")
        if pbi:
            # REDACCIÓN AUTOMÁTICA CON LENGUAJE ENDALIA
            redaccion = f"**Definición:**\nEsta funcionalidad permite al **colaborador** recibir avisos proactivos... \n\n**Configuración:**\nSe parametriza en el módulo de Planificación (Sección 5.2).\n\n**[Image {prox_img}]**"
            
            texto_final = st.text_area("Revisar redacción:", value=redaccion, height=250)
            
            st.markdown("---")
            st.markdown(f"### {nombre_nueva}")
            st.markdown(texto_final)
        else:
            st.info("Esperando PBI para redactar...")

    if pbi:
        if st.button("🚀 Generar y Fusionar Documento Completo"):
            with st.spinner("Fusionando con el documento de 80 páginas..."):
                doc = Document(archivo_maestro)
                
                # Buscamos el párrafo de la sección elegida para insertar justo después
                for i, p in enumerate(doc.paragraphs):
                    if p.text == ubicacion:
                        # Insertamos el nuevo título con estilo Heading 2
                        new_h = doc.paragraphs[i].insert_paragraph_before(nombre_nueva, style='Heading 2')
                        # Insertamos el contenido
                        new_p = doc.paragraphs[i].insert_paragraph_before(texto_final)
                        
                        if capturas:
                            doc.paragraphs[i].insert_paragraph_before(f"\n[Image {prox_img}]")
                            doc.add_picture(capturas[0], width=Inches(5))
                        break
                
                # Exportación
                buf = io.BytesIO()
                doc.save(buf)
                st.success("✅ ¡Fusión completada con éxito!")
                st.download_button("📥 Descargar Word Actualizado", buf.getvalue(), "Endalia_Funcional_UPDATE.docx")

else:
    st.info("Sube el archivo para que la IA detecte que la última imagen es la [Image 97].")
